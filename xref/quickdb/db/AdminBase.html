134     protected int saveProcess(ArrayList array, Object object) {
135         int index = -1;
136         String table = "";
137         int i = 2;
138         try {
139             if (this.forceTable) {
140                 table = tableForcedName;
141                 array.set(0, table);
142             } else {
143                 table = String.valueOf(array.get(0));
144             }
145 
146             conex.openBlock(table);
147 
148             int size = array.size() - 1;
149             for (; i < size; i++) {
150                 Object[] obj = (Object[]) array.get(i);
151 
152                 String column = (String) obj[0];
153                 Object value = obj[1];
154 
155                 conex.insertField(column, value);
156             }
157 
158             index = conex.closeBlock();
159             if (this.collection) {
160                 this.saveMany2Many(((String) array.get(0)), true, index);
161             }
162             this.manager.updateCache(object.getClass(), this);
163         } catch (SQLException ex) {
164             int value = -1;
165             try{
166                 this.conex.getConnection().setAutoCommit(true);
167                 if (!this.checkTableExist(table)) {
168                     if (this.createTable(object, array.toArray())) {
169                         value = this.saveProcess(array, object);
170                     }
171                 } else {
172                     if (this.addColumn(table, object, array, i)) {
173                         value = this.saveProcess(array, object);
174                     }
175                 }
176             }catch(Exception e){}
177             return value;
178         } catch (Exception e) {
179             try {
180                 conex.cancelTransaction();
181             } catch (Exception ex) {
182                 return -1;
183             }
184         }
185 
186         return index;
187     }

223     protected boolean saveMany2Many(String table, boolean saveNotModify, int index) {
224         ArrayList many = new ArrayList();
225         try {
226             String where = "";
227 
228             int sizeCollection = this.manager.sizeCollectionStack();
229             for (int i = 0; i < sizeCollection; i++) {
230                 ArrayList collec = this.manager.getCollection();
231                 int size = collec.size();
232                 if(size > 0 && collec.get(0) instanceof PrimitiveCollec){
233                     for (int q = 0; q < size; q++) {
234                         ((PrimitiveCollec) collec.get(q)).setBase(index);
235                         many.add(collec.get(q));
236                     }
237                 }else{
238                     for (int q = 0; q < size; q++) {
239                         many.add(new M2mTable(index,
240                                 ((Integer) collec.get(q))));
241                     }
242                 }
243 
244                 this.collection = false;
245                 this.forceTable = true;
246                 if (this.collectionHasName) {
247                     this.tableForcedName = this.manager.getNameCollection();
248                 } else {
249                     String nameCollection = this.manager.getNameCollection();
250                     //Consult if the table name or its inverse exists and asign match
251                     this.tableForcedName = table + nameCollection;
252                     where = "base = " + index;
253                     String tempTableName = nameCollection + table;
254                     if (!this.checkTableExist(this.tableForcedName) &&
255                             this.checkTableExist(tempTableName)) {
256                         this.tableForcedName = tempTableName;
257                         where = "related = " + index;
258                         for (M2mTable m2m : ((ArrayList<M2mTable>) many)) {
259                             int temp = m2m.getBase();
260                             m2m.setBase(m2m.getRelated());
261                             m2m.setRelated(temp);
262                         }
263                     }
264                 }
265 
266                 if (!saveNotModify) {
267                     String sql = "DELETE FROM " + this.tableForcedName + " WHERE " + where;
268                     this.executeQuery(sql);
269                 }
270                 this.saveAll(many);
271                 many.clear();
272             }
273         } catch (Exception e) {
274             return false;
275         } finally {
276             this.collection = false;
277             this.forceTable = false;
278             this.collectionHasName = false;
279         }
280 
281         return true;
282     }

285      * For this method only is the object to be modify.
286      * It is necessary to execute obtain previously to recover
287      * the proper object.
288      * @param Object
289      * @return True if the modification was successfull, False in the other case

291     public boolean modify(Object object) {
292         if(!this.manager.checkOptimisticLock(this, object)){
293             throw new OptimisticLockException();
294         }
295         boolean value = false;
296         try{
297             boolean commitValue = this.commit;
298             if(this.commit || this.autoCommit){
299                 conex.initTransaction();
300                 this.commit = false;
301             }
302 
303             ArrayList<String> manys = this.manager.getRef().isMany2Many(object.getClass());
304             if (!manys.isEmpty()) {
305                 if (this.manager.deleteMany2Many(this, object, manys)) {
306                     this.manager.cleanStack();
307                     value = this.save(object);
308                 }
309             } else {
310                 value = this.modifyProcess(object);
311             }
312             
313             this.commit = commitValue;
314             if(this.commit || this.autoCommit) conex.confirmTransaction();
315         }catch(SQLException e){
316             this.commit = true;
317         }
318 
319         return value;
320     }

322     protected boolean modifyProcess(Object object) {
323         ArrayList array = this.manager.entity2Array(this, object,
324                 EntityManager.OPERATION.MODIFY);
325 
326         try {
327             this.rs = conex.updateField(((String) array.get(0)),
328                     String.valueOf(array.get(array.size() - 1)));
329             rs.next();
330 
331             do {
332                 int size = array.size() - 1;
333                 for (int i = 1; i < size; i++) {
334                     Object[] obj = (Object[]) array.get(i);
335 
336                     String column = (String) obj[0];
337                     Object value = obj[1];
338 
339                     rs.updateObject(column, value);
340                     rs.updateRow();
341                 }
342             } while (rs.next());
343 
344             int index = Integer.parseInt(String.valueOf(array.get(array.size() - 1)).
345                     substring(String.valueOf(array.get(array.size() - 1)).indexOf("=")+1));
346 
347             if (this.collection) {
348                 this.saveMany2Many(((String) array.get(0)), false, index);
349             }
350             this.manager.updateCache(object.getClass(), this);
351         } catch (Exception e) {
352             try {
353                 conex.cancelTransaction();
354             } catch (Exception ex) {
355                 return false;
356             }
357             return false;
358         }
359 
360         return true;
361     }

363     private int modifyGetIndex(Object object) {
364         int index = this.manager.getRef().checkIndexValue(object);
365         /*If the index is greater than 0 it means that the Object already
366         exist in the Database and has to be modify, if index is lower than
367         0 it means that this is a new Object that has to be added to the
368         database*/
369         if (index > 0) {
370             this.modify(object);
371         } else {
372             index = this.saveGetIndex(object);
373         }
374 
375         return index;
376     }

390             }

394 
395             conex.deleteRows(((String) array.get(0)),
396                     ((String) array.get(array.size() - 1)));
397 
398             this.manager.deleteParent(this, object);
399 
400             //Obtain the attributes from this object that are collections
401             //and delete their relations in the Relational Table.
402             ArrayList<String[]> strings = this.manager.getCollectionsTableForDelete(object, this);
403             int index = (Integer) ((Object[]) array.get(1))[1];
404             for (String[] s : strings) {
405                 if (this.checkTableExist(s[0])) {
406                     this.executeQuery("DELETE FROM " + s[0] + " WHERE base=" + index);
407                 } else if (this.checkTableExist(s[1])) {
408                     this.executeQuery("DELETE FROM " + s[1] + " WHERE related=" + index);
409                 }
410             }
411 
412             this.commit = commitValue;
413             if(this.commit || this.autoCommit) conex.confirmTransaction();
414         } catch (Exception e) {
415             this.commit = true;
416             try {
417                 conex.cancelTransaction();
418             } catch (Exception ex) {
419                 return false;
420             }
421             return false;

458     /**
459      * Receive an Object and a query and return a Collection of object
460      * from the same type as was received, that represent the result of
461      * that query.
462      * @param Object
463      * @param Query [String]
464      * @return ArrayList [Collection of Object]
465      */
466     public ArrayList obtainAll(Class clazz, String sql) {
467         boolean cacheable = this.manager.isCacheable(clazz);
468         if(cacheable){
469             ArrayList array = this.manager.obtainCache(sql, this, clazz);
470             if(array != null){
471                 return array;
472             }
473         }
474         this.startObtainAll = true;
475         ArrayList results = new ArrayList();
476         try {
477             boolean value = false;
478             Object object = this.manager.getRef().emptyInstance(clazz);
479             if(sql.startsWith("SELECT ")){
480                 value = this.obtainSelect(object, sql);
481             }else{
482                 value = this.obtainWhere(object, sql);
483             }
484             if (value) {
485                 this.rs = this.arrayRs.peek();
486                 results.add(object);
487                 Object obj = this.manager.getRef().emptyInstance(object);
488                 while (this.obtainNext(obj)) {
489                     results.add(obj);
490                     obj = this.manager.getRef().emptyInstance(object);
491                 }
492             }
493             if(cacheable){
494                 this.manager.makeCacheable(sql, results, clazz, this);
495             }
496         } catch (Exception e) {
497             return null;
498         } finally {
499             this.startObtainAll = false;
500             this.arrayRs.pop();
501             if (!this.arrayRs.isEmpty()) {
502                 this.rs = this.arrayRs.peek();
503             }
504         }
505 
506         return results;
507     }

570     /**
571      * This Method return an Object[] containing String[] in each item,
572      * that represent the result of the specified query as a Matrix of Strings
573      * @param SQL Query [String]
574      * @param Number of Columns involved [int]
575      * @return Object[] or null
576      */
577     public Object[] obtainTable(Object... prop) {
578         try {
579             this.rs = conex.select((String)prop[0]);
580             rs.next();
581             int cols = 0;
582             String[] names = new String[0];
583             boolean hash = false;
584             if(prop.length > 1 && prop[1].getClass().getSimpleName().equalsIgnoreCase("Integer")){
585                 cols = (Integer) prop[1];
586             }else if(prop.length > 1 && prop[1].getClass().getSimpleName().equalsIgnoreCase("Boolean")){
587                 if((Boolean)prop[1]){
588                     String sql = ((String) prop[0]).toLowerCase();
589                     names = sql.substring(
590                             sql.indexOf("select ")+7, sql.indexOf(" from")).split(",");
591                     cols = names.length;
592                     hash = true;
593                 }else{
594                     String sql = ((String) prop[0]).toLowerCase();
595                     cols = sql.substring(
596                             sql.indexOf("select ")+7, sql.indexOf(" from")).split(",").length;
597                 }
598             }else{
599                 throw new InvalidParameterException();
600             }
601 
602             if(hash){
603                 ArrayList<Hashtable<String, Object>> array = new ArrayList<Hashtable<String, Object>>();
604                 do {
605                     Hashtable<String, Object> table = new Hashtable<String, Object>(cols);
606                     for (int i = 1; i < cols + 1; i++) {
607                         table.put(names[i-1].trim(), rs.getObject(i));
608                     }
609                     array.add(table);
610                 } while (rs.next());
611                 return array.toArray();
612             }else{
613                 ArrayList<String[]> array = new ArrayList<String[]>();
614                 do {
615                     String[] objs = new String[cols];
616                     for (int i = 1; i < cols + 1; i++) {
617                         objs[i - 1] = rs.getObject(i).toString();
618                     }
619                     array.add(objs);
620                 } while (rs.next());
621                 return array.toArray();
622             }
623 
624         } catch (Exception e) {
625             return null;
626         }
627     }

684      * Save all the items in the Collection in the Database
685      * @param An Object that implement java.util.Collection
686      * @return An ArrayList with the number of elements inserted or
687      * an ArrayList with each item index if the array received
688      * represent a Collection Reference.
689      */
690     public ArrayList<Integer> saveAll(Collection array) {
691         ArrayList<Integer> results = new ArrayList<Integer>();
692         try{
693             boolean commitValue = this.commit;
694             if(this.commit || this.autoCommit){
695                 conex.initTransaction();
696                 this.commit = false;
697             }
698             
699             Object objects[] = array.toArray();
700             if (!this.collection) {
701                 for (int i = 0; i < objects.length; i++) {
702                     if (!this.save(objects[i])) {
703                         results.add(i); //throw exception
704                         break;
705                     }
706                 }
707             } else {
708                 this.collection = false;
709                 for (Object obj : objects) {
710                     results.add(this.saveGetIndex(obj));
711                 }
712                 this.collection = true;
713             }
714 
715             this.commit = commitValue;
716             if(this.commit || this.autoCommit){
717                 conex.confirmTransaction();
718                 if(this.conex.getConnection().getAutoCommit()){
719                     this.conex.getConnection().setAutoCommit(false);
720                 }
721             }
722         }catch(SQLException e){
723             this.commit = true;
724         }
725 
726         return results;
727     }
